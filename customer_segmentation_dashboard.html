import pandas as pd
import numpy as np
from sklearn.preprocessing import StandardScaler
from sklearn.cluster import KMeans
from sklearn.decomposition import PCA
import matplotlib.pyplot as plt
import seaborn as sns

# 1. Load Data
df = pd.read_csv('customers.csv')
print(df.head(), df.describe())

# 2. Feature Engineering
features = ['age', 'income', 'spending_score', 'purchase_freq']
X = df[features].dropna()

# 3. Normalize
scaler = StandardScaler()
X_scaled = scaler.fit_transform(X)

# 4. Elbow Method — find optimal k
inertias = []
for k in range(1, 11):
    km = KMeans(n_clusters=k, random_state=42)
    km.fit(X_scaled)
    inertias.append(km.inertia_)

# 5. Fit K-Means with k=4
kmeans = KMeans(n_clusters=4, random_state=42, n_init=10)
df['Segment'] = kmeans.fit_predict(X_scaled)

# 6. Segment profiling
profile = df.groupby('Segment')[features].mean().round(1)
print(profile)

# 7. PCA for scatter visualization
pca = PCA(n_components=2)
X_pca = pca.fit_transform(X_scaled)
df['PC1'] = X_pca[:, 0]
df['PC2'] = X_pca[:, 1]

# 8. Plot clusters
plt.figure(figsize=(10, 6))
colors = ['#2d4a3e', '#8b3a52', '#1e3a5f', '#7a4f1e']
for seg, color in enumerate(colors):
    mask = df['Segment'] == seg
    plt.scatter(df[mask]['PC1'], df[mask]['PC2'], c=color, label=f'Segment {seg}', alpha=0.7)
plt.title('Customer Segments (PCA)')
plt.legend()
plt.show()

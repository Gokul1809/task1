# data_cleaning_pipeline.py — Task 4 Internship Project import pandas as pd import numpy as np from datetime import datetime # ── Step 1: Ingest ───────────────────────────────────────── def load_data(path): df = pd.read_csv(path, parse_dates=['date']) print(f"Loaded {len(df)} rows, {df.shape[1]} columns") return df # ── Step 2: Handle missing values ────────────────────────── def handle_missing(df): null_pct = df.isnull().mean() drop_cols = null_pct[null_pct > 0.5].index.tolist() df = df.drop(columns=drop_cols) num_cols = df.select_dtypes(include='number').columns df[num_cols] = df[num_cols].fillna(df[num_cols].median()) cat_cols = df.select_dtypes(include='object').columns df[cat_cols] = df[cat_cols].fillna(df[cat_cols].mode().iloc[0]) return df # ── Step 3: Remove duplicates ────────────────────────────── def remove_duplicates(df, key='order_id'): before = len(df) df = df.drop_duplicates(subset=[key], keep='first') print(f"Removed {before - len(df)} duplicate rows") return df # ── Step 4: Standardise ──────────────────────────────────── def standardise(df): df['region'] = df['region'].str.strip().str.title() df['status'] = df['status'].str.lower().str.strip() df['amount'] = pd.to_numeric( df['amount'].astype(str).str.replace(r'[₹$,]', '', regex=True), errors='coerce') return df # ── Step 5: Cap outliers (IQR method) ───────────────────── def cap_outliers(df, col='amount'): Q1 = df[col].quantile(0.25) Q3 = df[col].quantile(0.75) IQR = Q3 - Q1 lo, hi = Q1 - 1.5*IQR, Q3 + 1.5*IQR df[col] = df[col].clip(lo, hi) return df # ── Step 6: Quality report ───────────────────────────────── def quality_report(df): completeness = (1 - df.isnull().mean()).mean() * 100 print(f"Completeness: {completeness:.1f}%") df.to_csv('cleaned_data.csv', index=False) print("Saved → cleaned_data.csv") # ── Run pipeline ─────────────────────────────────────────── if __name__ == '__main__': df = load_data('raw_sales.csv') df = handle_missing(df) df = remove_duplicates(df) df = standardise(df) df = cap_outliers(df) quality_report(df)
